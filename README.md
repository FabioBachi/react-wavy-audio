```bash
npm install --save react-wavy-audio
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-wavy-audio'
import 'react-wavy-audio/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
